Transitions

D3 transitions are easy, powerful, and neat. Aside from being pretty transitions are useful for reorienting the user when something changes.

var data = [0,6,2,4,9,8,1,3,5,3,2,6,8,1,7];  

// Create circle selection
var circle = d3.select("#canvas")
  .append("svg")
  .selectAll("circle")
  .data(data);

// Initial circle state
circle
  .enter()
  .append("circle")
  .attr("r", 10)
  .attr("cx", function(d,i) {
    return (i+1) * 50;
  })
  .attr("cy", 40)
  .attr("fill", "#3498db")

  // Rescale radius, go purple
  .transition()
  .delay(function(d,i) {
    return 500 + 25*i;
  })
  .duration(1000)
  .attr("r", function(d,i) {
    return d * 3;
  })
  .attr("fill", "#9b59b6")

  // Shrink, go red
  .transition()
  .duration(250)
  .attr("fill", "#e74c3c")
  .attr("r", 0)

  // Unshrink, go orange
  .transition()
  .duration(250)
  .attr("fill", "#e67e22")
  .attr("r", 10)

  // Rise, go yellow
  .transition()
  .duration(1000)
  .attr("fill", "#f1c40f")
  .attr("r", 15)
  .attr("cy", 20)

  // Fade out
  .transition()
  .duration(1000)
  .style("opacity", 0)
  .attr("cy", 100)

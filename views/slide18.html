A Simple Barchart

We're ready to add some bars now.

svg.selectAll("rect")
  .data(data)
  .enter()
  .append("rect")
  .attr("x", function(d) {
    return x(d.key)
  })
  .attr("width", x.rangeBand())
  .attr("y", function(d) {
    return y(d.values.length);
  })
  .attr("height", function(d) {
    return height - y(d.values.length)
  });
